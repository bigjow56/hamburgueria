    <script>
        // Switch between login and register forms
        function switchTab(tab) {
            const loginTab = document.querySelector('.tab-btn:first-child');
            const registerTab = document.querySelector('.tab-btn:last-child');
            const loginForm = document.getElementById('loginForm');
            const registerForm = document.getElementById('registerForm');

            if (tab === 'login') {
                loginTab.classList.add('active');
                registerTab.classList.remove('active');
                loginForm.style.display = 'block';
                registerForm.style.display = 'none';
            } else {
                registerTab.classList.add('active');
                loginTab.classList.remove('active');
                loginForm.style.display = 'none';
                registerForm.style.display = 'block';
            }

            hideAlert();
        }

        // Show/hide alert messages
        function showAlert(message, type = 'error') {
            const alert = document.getElementById('alert');
            alert.textContent = message;
            alert.className = `alert ${type}`;
            alert.style.display = 'block';
        }

        function hideAlert() {
            document.getElementById('alert').style.display = 'none';
        }

        // Show loading state
        function setLoading(button, isLoading) {
            const btnText = button.querySelector('.btn-text');
            const loading = button.querySelector('.loading');
            
            if (isLoading) {
                btnText.style.opacity = '0';
                loading.style.display = 'block';
                button.disabled = true;
            } else {
                btnText.style.opacity = '1';
                loading.style.display = 'none';
                button.disabled = false;
            }
        }

        // Handle login form submission
        document.getElementById('loginForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            hideAlert();

            const button = this.querySelector('.login-btn');
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;

            if (!email || !password) {
                showAlert('Preencha todos os campos!');
                return;
            }

            setLoading(button, true);

            try {
                // Simular chamada da API
                await new Promise(resolve => setTimeout(resolve, 1500));

                // Aqui você faria a chamada real para sua API
                const response = await fetch('/api/auth/login', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ email, password })
                });

                if (response.ok) {
                    const data = await response.json();
                    showAlert('Login realizado com sucesso!', 'success');
                    
                    // Salvar token (se necessário)
                    localStorage.setItem('token', data.token);
                    
                    // Redirecionar após 1 segundo
                    setTimeout(() => {
                        window.location.href = '/dashboard.html';
                    }, 1000);
                } else {
                    const error = await response.json();
                    showAlert(error.message || 'Email ou senha incorretos!');
                }
            } catch (error) {
                // Para demonstração - remover em produção
                if (email === 'demo@teste.com' && password === '123456') {
                    showAlert('Login realizado com sucesso!', 'success');
                    setTimeout(() => {
                        alert('Redirecionando para o dashboard...');
                    }, 1000);
                } else {
                    showAlert('Email ou senha incorretos!');
                }
            } finally {
                setLoading(button, false);
            }
        });

        // Handle register form submission
        document.getElementById('registerForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            hideAlert();

            const button = this.querySelector('.login-btn');
            const name = document.getElementById('registerName').value;
            const email = document.getElementById('registerEmail').value;
            const password = document.getElementById('registerPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;

            if (!name || !email || !password || !confirmPassword) {
                showAlert('Preencha todos os campos!');
                return;
            }

            if (password !== confirmPassword) {
                showAlert('As senhas não coincidem!');
                return;
            }

            if (password.length < 6) {
                showAlert('A senha deve ter pelo menos 6 caracteres!');
                return;
            }

            setLoading(button, true);

            try {
                // Simular chamada da API
                await new Promise(resolve => setTimeout(resolve, 2000));

                // Aqui você faria a chamada real para sua API
                const response = await fetch('/api/auth/register', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ name, email, password })
                });

                if (response.ok) {
                    showAlert('Conta criada com sucesso! Você ganhou 100 pontos de bônus!', 'success');
                    
                    // Limpar formulário
                    this.reset();
                    
                    // Voltar para login após 2 segundos
                    setTimeout(() => {
                        switchTab('login');
                    }, 2000);
                } else {
                    const error = await response.json();
                    showAlert(error.message || 'Erro ao criar conta. Tente novamente.');
                }
            } catch (error) {
                // Para demonstração - sempre sucesso
                showAlert('Conta criada com sucesso! Você ganhou 100 pontos de bônus!', 'success');
                this.reset();
                setTimeout(() => {
                    switchTab('login');
                }, 2000);
            } finally {
                setLoading(button, false);
            }
        });

        // Handle forgot password
        function showForgotPassword() {
            const contact = prompt('Digite seu email ou telefone para recuperar a senha:');
            if (contact) {
                const isEmail = contact.includes('@');
                const message = isEmail ? 
                    'Instruções enviadas para seu email!' : 
                    'Código de recuperação enviado por SMS!';
                showAlert(message, 'success');
            }
        }

        // Auto-focus first input when switching tabs
        document.addEventListener('DOMContentLoaded', function() {
            document.getElementById('loginContact').focus();
        });

        // Aplicar máscara de telefone no campo de cadastro
        document.getElementById('registerPhone').addEventListener('input', function(e) {
            let value = e.target.value.replace(/\D/g, '');
            
            if (value.length <= 10) {
                value = value.replace(/(\d{2})(\d{4})(\d{0,4})/, '($1) $2-$3');
            } else {
                value = value.replace(/(\d{2})(\d{5})(\d{0,4})/, '($1) $2-$3');
            }
            
            e.target.value = value.trim();
        });
    </script>
